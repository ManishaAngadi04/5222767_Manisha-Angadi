//Problem 7 - Diagonal Difference

#include<stdio.h>
#include<stdlib.h>

int diagonalDifference(int arr[][100], int n){
    int primaryDiagonal = 0;
    int secondaryDiagonal = 0;
    
    for(int i=0;i<n;i++){
        primaryDiagonal += arr[i][i];
        secondaryDiagonal += arr[i][n-1-i];
    }
    return abs(primaryDiagonal - secondaryDiagonal);
}
int main(){
    int n;
    scanf("%d", &n);
    int arr[100][100];
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            scanf("%d",&arr[i][j]);
        }
    }
    int result = diagonalDifference(arr,n);
    printf("%d\n", result);
    return 0;
}
