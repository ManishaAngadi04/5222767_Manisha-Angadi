//Problem 10 - Queue using Two Stacks

#include<stdio.h>

int main(){
    int q,op,x;
    scanf("%d",&q);
    int a[1000000],front=0,rear=0;
    
    while(q--){
        scanf("%d",&op);
        if(op==1) scanf("%d",&a[rear++]);
        else if(op==2) front++;
        else printf("%d\n",a[front]);
    }
    return 0;
}
