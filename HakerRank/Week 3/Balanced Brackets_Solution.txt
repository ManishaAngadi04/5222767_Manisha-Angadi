//Problem 11 - Balanced Brackets

#include<stdio.h>
#include<string.h>

char* isBalanced(char* s){
    char stack[1005];
    int top=-1;
    for(int i=0;s[i];i++){
        char c=s[i];
        if(c=='('||c=='{'||c=='[') stack[++top]=c;
        else{
            if(top==-1) return "NO";
            char open=stack[top--];
            if((c==')'&&open!='(')||(c=='}'&&open!='{')||(c==']'&&open!='['))
            return "NO";
        }
    }
    return top == -1?"YES":"NO";
}
int main(){
    int n;
    scanf("%d",&n);
    char s[1005];
    while(n--){
        scanf("%s",s);
        printf("%s\n",isBalanced(s));
    }
    return 0;
}
