//Problem 2 - New Year Chaos

#include<stdio.h>

void minimumBrides(int q[],int n){
    int brides=0;
    for(int i=0;i<n;i++){
        if(q[i]-(i+1)>2){
            printf("Too chaotic\n");
            return;
        }
        for(int j=(q[i]-2>0?q[i]-2:0); j<i;j++){
            if(q[j]>q[i]){
                brides++;
            }
        }
    }
    printf("%d\n",brides);
}
int main(){
    int t;
    scanf("%d", &t);
    while(t--){
        int n;
        scanf("%d",&n);
        int q[n];
        for(int i=0;i<n;i++){
            scanf("%d", &q[i]);
        }
        minimumBrides(q, n);
    }
    return 0;
}
