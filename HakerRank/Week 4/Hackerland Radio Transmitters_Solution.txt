//Problem 6 - Hackerland Radio Transmitters

#include <stdio.h>
#include <stdlib.h>

int cmp(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int main() {
    int n, k;
    scanf("%d %d", &n, &k);
    int houses[n];
    for (int i = 0; i < n; i++) scanf("%d", &houses[i]);
    qsort(houses, n, sizeof(int), cmp);
    int transmitters = 0, i = 0;
    while (i < n) {
        transmitters++;
        int loc = houses[i] + k;
        while (i < n && houses[i] <= loc) i++;
        int pos = houses[i - 1] + k;
        while (i < n && houses[i] <= pos) i++;
    }
    printf("%d\n", transmitters);
    return 0;
}
